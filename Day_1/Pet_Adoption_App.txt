CLASS Pet
    PROPERTIES:
        name
        age
        breed
    
    CONSTRUCTOR(name, age, breed)
        this.name  = name
        this.age   = age
        this.breed = breed
    
    METHOD displayInfo()
        PRINT "Name: " + name
        PRINT "Age: " + age
        PRINT "Breed: " + breed


CLASS Dog EXTENDS Pet
    PROPERTIES:
        isTrained
    
    CONSTRUCTOR(name, age, breed, isTrained)
        CALL Pet.CONSTRUCTOR(name, age, breed)
        this.isTrained = isTrained
    
    METHOD displayInfo()
        CALL super.displayInfo()
        PRINT "Trained: " + isTrained
