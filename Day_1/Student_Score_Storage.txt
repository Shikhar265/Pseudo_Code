MAIN PROGRAM:
    CREATE marksMap as HashMap<String, Integer>

    // Step 1: Store student scores
    PUT ("Alice", 85) into marksMap
    PUT ("Bob", 92) into marksMap
    PUT ("Charlie", 78) into marksMap
    PUT ("Diana", 95) into marksMap

    // Step 2: Find highest scorer using streams
    highestEntry = marksMap.entrySet()
                           .stream()
                           .max(compare by value)
                           .get()

    PRINT "Highest Scorer: " + highestEntry.key + " with score " + highestEntry.value
END PROGRAM
